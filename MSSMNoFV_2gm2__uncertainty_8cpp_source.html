34 /**
35  * Calculates uncertainty associated with amu(0-loop) including
36  * tan(beta) resummation.
37  *
38  * The estimated uncertainty is the magnitude amu(1-loop) (including
39  * tan(beta) resummation).
40  *
41  * @param model model parameters
42  *
43  * @return uncertainty for amu(0-loop) w/ tan(beta) resummation
44  */
45 double calculate_uncertainty_amu_0loop(const MSSMNoFV_onshell& model)
46 {
47  const double amu_1L = calculate_amu_1loop(model);
48 
49  return std::abs(amu_1L);
50 }

52 /**
53  * Calculates uncertainty associated with amu(1-loop) including
54  * tan(beta) resummation.
55  *
56  * The estimated uncertainty is the sum of magnitude amu(2-loop)
57  * (including tan(beta) resummation) and the 2-loop uncertainty,
58  * calculated by calculate_uncertainty_amu_2loop().
59  *
60  * @param model model parameters
61  *
62  * @return uncertainty for amu(1-loop) w/ tan(beta) resummation
63  */
64 double calculate_uncertainty_amu_1loop(const MSSMNoFV_onshell& model)
65 {
66  const double amu_2L = calculate_amu_2loop(model);
67  const double delta_amu_2L = calculate_uncertainty_amu_2loop(model);
68 
69  return std::abs(amu_2L) + std::abs(delta_amu_2L);
70 }

72 /**
73  * Calculates uncertainty associated with amu(2-loop) using Eq (4).
74  *
75  * Eq. (4) takes into account the unknown two-loop contributions and
76  * the employed approximation for the 2L(a) contributions.
77  *
78  * @param model model parameters
79  *
80  * @return uncertainty for amu(2-loop)
81  */
82 double calculate_uncertainty_amu_2loop(const MSSMNoFV_onshell& model)
83 {
84  const double amu_2La_Cha = amu2LaCha(model);
85  const double amu_2La_Sferm = amu2LaSferm(model);
86 
87  return 2.3e-10 + 0.3 * (std::abs(amu_2La_Cha) + std::abs(amu_2La_Sferm));
88 }
