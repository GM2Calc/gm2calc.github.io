43/// fast implementation of complex logarithm
44template <class T>
45std::complex<T> log(const std::complex<T>& z) noexcept
46{
47 const T rz = std::real(z);
48 const T iz = std::imag(z);
49
50 return std::complex<T>(0.5*std::log(rz*rz + iz*iz), std::atan2(iz, rz));
51}
