136 /**
137  * Fills a matrix from an SLHA block
138  *
139  * @param block the block
140  * @param matrix matrix to be filled
141  */
142 template <class Derived>
143 void GM2_slha_io::read_matrix(const SLHAea::Block& block, Eigen::MatrixBase<Derived>& matrix)
144 {
145  using Index_t = Eigen::Index;
146  static_assert(std::is_signed<Index_t>::value, "Eigen::Index must be a signed integer type.");
147 
148  const Index_t cols = matrix.cols(), rows = matrix.rows();
149 
150  for (const auto& line : block) {
151  if (line.is_data_line() && line.size() >= 3) {
152  const Index_t i = convert_to<Index_t>(line[0]) - 1;
153  const Index_t k = convert_to<Index_t>(line[1]) - 1;
154  if (0 <= i && i < rows && 0 <= k && k < cols) {
155  matrix(i, k) = convert_to<double>(line[2]);
156  }
157  }
158  }
159 }

161 /**
162  * Fills a vector from an SLHA block
163  *
164  * @param block the block
165  * @param vector vector to be filled
166  */
167 template <class Derived>
168 void GM2_slha_io::read_vector(const SLHAea::Block& block, Eigen::MatrixBase<Derived>& vector)
169 {
170  using Index_t = Eigen::Index;
171  static_assert(std::is_signed<Index_t>::value, "Eigen::Index must be a signed integer type.");
172 
173  const Index_t rows = vector.rows();
174 
175  for (const auto& line : block) {
176  if (line.is_data_line() && line.size() >= 2) {
177  const Index_t i = convert_to<Index_t>(line[0]) - 1;
178  if (0 <= i && i < rows) {
179  vector(i) = convert_to<double>(line[1]);
180  }
181  }
182  }
183 }

193 {

201 /**
202  * Fills a matrix from a SLHA block
203  *
204  * @param block_name block name
205  * @param matrix matrix to be filled
206  * @param scale (or 0 if scale should be ignored)
207  */
208 template <class Derived>
209 void GM2_slha_io::read_block(const std::string& block_name,
210  Eigen::MatrixBase<Derived>& matrix,
211  double scale) const
212 {
213  auto block = SLHAea::Coll::find(data.cbegin(), data.cend(), block_name);
214 
215  while (block != data.cend()) {
216  if (is_at_scale(*block, scale)) {
217  GM2_slha_io::read_block(*block, matrix);
218  }
219 
220  ++block;
221  block = SLHAea::Coll::find(block, data.cend(), block_name);
222  }
223 }
