28template <typename Derived>
29unsigned closest_index(double mass, const Eigen::ArrayBase<Derived>& v)
30{
31 unsigned pos = 0;
32 typename Derived::PlainObject tmp;
33 tmp.setConstant(mass);
34
35 (v - tmp).abs().minCoeff(&pos);
36
37 return pos;
38}

54/**
55 * Normalize each element of the given real matrix to be within the
56 * interval [min, max]. Values < min are set to min. Values > max
57 * are set to max.
58 *
59 * @param m matrix
60 * @param min minimum
61 * @param max maximum
62 */
63template <int M, int N>
64void normalize_to_interval(Eigen::Matrix<double,M,N>& m, double min = -1., double max = 1.)
65{
66 auto data = m.data();
67 const auto size = m.size();
68
69 for (int i = 0; i < size; i++) {
70 if (data[i] < min)
71 data[i] = min;
72 else if (data[i] > max)
73 data[i] = max;
74 }
75}

77/**
78 * The element of v, which is closest to mass, is moved to the
79 * position idx.
80 *
81 * @param idx new index of the mass eigenvalue
82 * @param mass mass to compare against
83 * @param v vector of masses
84 * @param z corresponding mixing matrix
85 */
86template <typename DerivedArray, typename DerivedMatrix>
87void move_goldstone_to(int idx, double mass, Eigen::ArrayBase<DerivedArray>& v,
88 Eigen::MatrixBase<DerivedMatrix>& z)
89{
90 int pos = closest_index(mass, v);
91 if (pos == idx) {
92 return;
93 }
94
95 const int sign = (idx - pos) < 0 ? -1 : 1;
96 int steps = std::abs(idx - pos);
97
98 // now we shuffle the states
99 while (steps--) {
100 const int new_pos = pos + sign;
101 v.row(new_pos).swap(v.row(pos));
102 z.row(new_pos).swap(z.row(pos));
103 pos = new_pos;
104 }
105}

107/**
108 * Returns all elements from src, which are not close to the elements
109 * in cmp. The returned vector will have the length (src.size() -
110 * cmp.size()).
111 *
112 * @param src source vector
113 * @param cmp vector with elements to compare against
114 * @return vector with elements of src not close to cmp
115 */
116template<class Real, int Nsrc, int Ncmp>
117Eigen::Array<Real,Nsrc - Ncmp,1> remove_if_equal(
118 const Eigen::Array<Real,Nsrc,1>& src,
119 const Eigen::Array<Real,Ncmp,1>& cmp)
120{
121 Eigen::Array<Real,Nsrc,1> non_equal(src);
122 Eigen::Array<Real,Nsrc - Ncmp,1> dst;
123
124 for (int i = 0; i < Ncmp; i++) {
125 const int idx = closest_index(cmp(i), non_equal);
126 non_equal(idx) = std::numeric_limits<double>::infinity();
127 }
128
129 std::remove_copy_if(non_equal.data(), non_equal.data() + Nsrc,
130 dst.data(), [] (auto x) { return !std::isfinite(x); });
131
132 return dst;
133}

135/**
136 * @brief reorders vector v according to ordering in vector v2
137 * @param v vector with elementes to be reordered
138 * @param v2 vector with reference ordering
139 */
140template<class Real, int N>
141void reorder_vector(
142 Eigen::Array<Real,N,1>& v,
143 const Eigen::Array<Real,N,1>& v2)
144{
145 Eigen::PermutationMatrix<N> p;
146 p.setIdentity();
147 std::sort(p.indices().data(), p.indices().data() + p.indices().size(),
148 [&v2] (int i, int j) { return std::abs(v2[i]) < std::abs(v2[j]); });
149
150#if EIGEN_VERSION_AT_LEAST(3,1,4)
151 v.matrix().transpose() *= p.inverse();
152#else
153 Eigen::Map<Eigen::Matrix<Real,N,1> >(v.data()).transpose() *= p.inverse();
154#endif
155}

158 * @brief reorders vector v according to ordering of diagonal elements in mass_matrix
159 * @param v vector with elementes to be reordered
160 * @param matrix matrix with diagonal elements with reference ordering

162template<class Derived>
163void reorder_vector(
164 Eigen::Array<double,Eigen::MatrixBase<Derived>::RowsAtCompileTime,1>& v,
165 const Eigen::MatrixBase<Derived>& matrix)
166{
167 reorder_vector(v, matrix.diagonal().array().eval());

170template <typename Derived>
171void symmetrize(Eigen::MatrixBase<Derived>& m)
172{
173 static_assert(Eigen::MatrixBase<Derived>::RowsAtCompileTime ==
174 Eigen::MatrixBase<Derived>::ColsAtCompileTime,
175 "symmetrize is only defined for squared matrices");
176
177 for (int i = 0; i < Eigen::MatrixBase<Derived>::RowsAtCompileTime; i++) {
178 for (int k = 0; k < i; k++) {
179 m(i,k) = m(k,i);
180 }
181 }
182}
