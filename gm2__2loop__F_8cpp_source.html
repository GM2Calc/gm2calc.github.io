70 double calc_v2(const THDM_F_parameters& thdm) noexcept
71 {
72  const double mw2 = sqr(thdm.mw);
73  const double mz2 = sqr(thdm.mz);
74  const double sw2 = 1.0 - mw2/mz2;
75  const double e2 = 4*pi*thdm.alpha_em;
76  const double g22 = e2/sw2;
77  return 4*mw2/g22;
78 }

80 /// Eq (56), arxiv:1607.06292, S = h or H
81 double FS(double ms2, double mf2) noexcept
82 {
83  if (std::abs(ms2 - 4*mf2) <= 2*std::numeric_limits<double>::epsilon()) {
84  return -2.0;
85  }
86 
87  return -2.0 + std::log(ms2/mf2)
88  - (ms2 - 2*mf2)/ms2 * Phi(ms2, mf2, mf2)/(ms2 - 4*mf2);
89 }

91 /// Eq (57), arxiv:1607.06292, S = A
92 double FA(double ms2, double mf2) noexcept
93 {
94  if (std::abs(ms2 - 4*mf2) <= 2*std::numeric_limits<double>::epsilon()) {
95  return 2.7725887222397812; // Log[16]
96  }
97 
98  return Phi(ms2, mf2, mf2)/(ms2 - 4*mf2);
99 }

102 template <typename F>
103 double fSgamma(double ms2, double mf2, const F_neut_pars& pars, F FS) noexcept
104 {
105  const double qf2 = sqr(pars.qf);
106  const double nc = pars.nc;
107 
108  return qf2*nc * mf2/ms2 * FS(ms2, mf2);

111 /// Eq (55), arxiv:1607.06292, S = h or H
112 template <typename F>
113 double fSZ(double ms2, double mf2, const F_neut_pars& pars, const F_sm_pars& sm, F FS) noexcept
114 {
115  const double mw2 = sm.mw2;
116  const double mz2 = sm.mz2;
117  const double cw2 = mw2/mz2;
118  const double sw2 = 1.0 - cw2;
119  const double qf = pars.qf;
120  const double ql = pars.ql;
121  const double nc = pars.nc;
122  const double gvf = 0.5*pars.t3f - qf*sw2;
123  const double gvl = 0.5*pars.t3l - ql*sw2;
124 
125  return (-nc*qf*gvl*gvf)/(sw2*cw2)
126  * mf2/(ms2 - mz2) * (FS(ms2, mf2) - FS(mz2, mf2));
127 }

137 double FlHp(double ms2, double mf2) noexcept
138 {

145 /**
146  * Eq (61), arxiv:1607.06292
147  *
148  * @note There is a misprint in Eq (61), arxiv:1607.06292v2: There
149  * should be no Phi function in the 2nd line of (61).
150  *
151  * @param phi Phi(xd, xu, 1.0)
152  */
153 double FdHp(double ms2, double md2, double mu2, double qd, double qu, double phi) noexcept
154 {
155  const double xu = mu2/ms2;
156  const double xd = md2/ms2;
157  const double y = sqr(xu - xd) - 2*(xu + xd) + 1.0;
158  const double s = 0.25*(qu + qd);
159  const double c = sqr(xu - xd) - qu*xu + qd*xd;
160  const double cbar = (xu - qu)*xu - (xd + qd)*xd;
161  const double lxu = std::log(xu);
162  const double lxd = std::log(xd);
163 
164  return -(xu - xd) + (cbar/y - c*(xu - xd)/y) * phi
165  + c*(dilog(1.0 - xd/xu) - 0.5*lxu*(lxd - lxu))
166  + (s + xd)*lxd + (s - xu)*lxu;
167 }

169 double FdHp(double ms2, double md2, double mu2, double qd, double qu) noexcept
170 {
171  const double xu = mu2/ms2;
172  const double xd = md2/ms2;
173 
174  return FdHp(ms2, md2, mu2, qd, qu, Phi(xd, xu, 1.0));
175 }

177 /// Eq (62), arxiv:1607.06292
178 double FuHp(double ms2, double md2, double mu2, double qd, double qu) noexcept
179 {
180  const double xu = mu2/ms2;
181  const double xd = md2/ms2;
182  const double y = sqr(xu - xd) - 2*(xu + xd) + 1.0;
183  const double phi = Phi(xd, xu, 1.0);
184 
185  return FdHp(ms2, md2, mu2, 2.0 + qd, 2.0 + qu, phi)
186  - 4.0/3.0*(xu - xd - 1.0)/y*phi
187  - 1.0/3.0*(sqr(std::log(xd)) - sqr(std::log(xu)));
188 }

309 /**
310  * Calculates 2-loop fermionic contributions with charged Higgs
311  * boson.
312  *
313  * Eq (52), arXiv:2110.13238
314  */
315 double amu2L_F_charged(const THDM_F_parameters& thdm) noexcept
316 {
317  const double mHp2 = sqr(thdm.mHp);
318  const double mw2 = sqr(thdm.mw);
319  const double mz2 = sqr(thdm.mz);
320  const double v2 = calc_v2(thdm);
321  const double sw2 = 1 - mw2/mz2;
322  const double pref = sqr(thdm.alpha_em*thdm.mm/(8*pi*thdm.mw*sw2))*v2/thdm.ml(1);
323 
324  double res = 0;
325 
326  // loop over generations
327  for (int i = 0; i < 3; ++i) {
328  for (int j = 0; j < 3; ++j) {
329  // u
330  res += fuHp(mHp2, sqr(thdm.md(j)), sqr(thdm.mu(i)), mw2, mz2)
331  *std::real(std::conj(thdm.yuHp(i,j))*thdm.vckm(i,j)*thdm.ylHp(1,1))/thdm.mu(i);
332  // d
333  res += fdHp(mHp2, sqr(thdm.md(j)), sqr(thdm.mu(i)), mw2, mz2)
334  *std::real(std::conj(thdm.ydHp(i,j))*thdm.vckm(i,j)*thdm.ylHp(1,1))/thdm.md(j);
335  }
336  // l
337  res += flHp(mHp2, sqr(thdm.ml(i)), mw2, mz2)
338  *std::real(std::conj(thdm.ylHp(i,i))*thdm.ylHp(1,1))/thdm.ml(i);
339  }
340 
341  return pref*res;
342 }

344 /**
345  * Calculates 2-loop fermionic contributions with neutral Higgs
346  * bosons.
347  *
348  * Eq (53), arxiv:1607:06292
349  */
350 double amu2L_F_neutral(const THDM_F_parameters& thdm) noexcept
351 {
352  const double mh2 = sqr(thdm.mh(0));
353  const double mH2 = sqr(thdm.mh(1));
354  const double mA2 = sqr(thdm.mA);
355  const double mhSM2 = sqr(thdm.mhSM);
356  const double mw2 = sqr(thdm.mw);
357  const double mz2 = sqr(thdm.mz);
358  const double v2 = calc_v2(thdm);
359  const double sw2 = 1 - mw2/mz2;
360  const auto pref = sqr(thdm.alpha_em*thdm.mm/(2*pi*thdm.mw))/sw2;
361 
362  double res = 0;
363 
364  // loop over generations
365  for (int i = 0; i < 3; ++i) {
366  // h
367  res += fuS(mh2, sqr(thdm.mu(i)), mw2, mz2)*std::real(std::conj(thdm.yuh(i,i))*thdm.ylh(1,1))*v2/(thdm.mu(i)*thdm.ml(1));
368  res += fdS(mh2, sqr(thdm.md(i)), mw2, mz2)*std::real(std::conj(thdm.ydh(i,i))*thdm.ylh(1,1))*v2/(thdm.md(i)*thdm.ml(1));
369  res += flS(mh2, sqr(thdm.ml(i)), mw2, mz2)*std::real(std::conj(thdm.ylh(i,i))*thdm.ylh(1,1))*v2/(thdm.ml(i)*thdm.ml(1));
370 
371  // H
372  res += fuS(mH2, sqr(thdm.mu(i)), mw2, mz2)*std::real(std::conj(thdm.yuH(i,i))*thdm.ylH(1,1))*v2/(thdm.mu(i)*thdm.ml(1));
373  res += fdS(mH2, sqr(thdm.md(i)), mw2, mz2)*std::real(std::conj(thdm.ydH(i,i))*thdm.ylH(1,1))*v2/(thdm.md(i)*thdm.ml(1));
374  res += flS(mH2, sqr(thdm.ml(i)), mw2, mz2)*std::real(std::conj(thdm.ylH(i,i))*thdm.ylH(1,1))*v2/(thdm.ml(i)*thdm.ml(1));
375 
376  // A
377  res += fuA(mA2, sqr(thdm.mu(i)), mw2, mz2)*std::real(std::conj(thdm.yuA(i,i))*thdm.ylA(1,1))*v2/(thdm.mu(i)*thdm.ml(1));
378  res += fdA(mA2, sqr(thdm.md(i)), mw2, mz2)*std::real(std::conj(thdm.ydA(i,i))*thdm.ylA(1,1))*v2/(thdm.md(i)*thdm.ml(1));
379  res += flA(mA2, sqr(thdm.ml(i)), mw2, mz2)*std::real(std::conj(thdm.ylA(i,i))*thdm.ylA(1,1))*v2/(thdm.ml(i)*thdm.ml(1));
380 
381  // subtract hSM
382  res -= fuS(mhSM2, sqr(thdm.mu(i)), mw2, mz2);
383  res -= fdS(mhSM2, sqr(thdm.md(i)), mw2, mz2);
384  res -= flS(mhSM2, sqr(thdm.ml(i)), mw2, mz2);
385  }
386 
387  return pref*res;
388 }
