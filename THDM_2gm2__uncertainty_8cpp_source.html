36 /**
37  * Calculates uncertainty associated with amu(0-loop)
38  *
39  * The estimated uncertainty is the magnitude amu(1-loop) plus
40  * amu(2-loop).
41  *
42  * @param model model parameters
43  *
44  * @return uncertainty for amu(0-loop)
45  */
46 double calculate_uncertainty_amu_0loop(const THDM& model)
47 {
48  const double amu_1L = calculate_amu_1loop(model);
49  const double amu_2L = calculate_amu_2loop(model);
50 
51  return std::abs(amu_1L) + std::abs(amu_2L);
52 }

54 /**
55  * Calculates uncertainty associated with amu(1-loop)
56  *
57  * The estimated uncertainty is the sum of magnitude amu(2-loop) and
58  * the 2-loop uncertainty, calculated by
59  * calculate_uncertainty_amu_2loop().
60  *
61  * @param model model parameters
62  *
63  * @return uncertainty for amu(1-loop)
64  */
65 double calculate_uncertainty_amu_1loop(const THDM& model)
66 {
67  const double amu_2L = calculate_amu_2loop(model);
68  const double delta_amu_2L = calculate_uncertainty_amu_2loop(model);
69 
70  return std::abs(amu_2L) + std::abs(delta_amu_2L);
71 }

73 /**
74  * Calculates uncertainty associated with amu(2-loop)
75  *
76  * Takes into account the neglected two-loop contribution
77  * \f$a_\mu^{\Delta r\text{-shift}}\f$, Eq.(34) arxiv:1607.06292,
78  * two-loop contributions of \f$O(m_\mu^4)\f$ and three-loop
79  * contributions of \f$O(m_\mu^2)\f$
80  *
81  * @param model model parameters
82  *
83  * @return uncertainty for amu(2-loop)
84  */
85 double calculate_uncertainty_amu_2loop(const THDM& model)
86 {
87  const double pi = 3.1415926535897932;
88  const double alpha_em = model.get_alpha_em();
89  const double mm = model.get_MFe(1);
90  const double mH = std::abs(model.get_Mhh(1));
91  const double mA = std::abs(model.get_MAh(1));
92  const double mHp = std::abs(model.get_MHm(1));
93  const double mNP = std::fmin(mH,std::fmin(mA,mHp)); // new physics scale
94 
95  // universal 2-loop QED logarithmic correction from Eq.(51) hep-ph/9803384
96  const double delta_alpha_em = -4*alpha_em/pi*std::log(std::abs(mNP/mm));
97 
98  // upper bound on neglected 2-loop contribution from \Delta r
99  const double delta_amu_2L_delta_r = 2e-12;
100 
101  // estimate of 2-loop corrections O(mm^4)
102  const double delta_amu_2L_mm4 = std::abs(calculate_amu_1loop(model)*delta_alpha_em);
103 
104  // estimate of 3-loop corrections O(mm^2)
105  const double delta_amu_3L = std::abs(calculate_amu_2loop(model)*delta_alpha_em);
106 
107  return delta_amu_2L_delta_r + delta_amu_2L_mm4 + delta_amu_3L;
108 }
