46 /**
47  * Calculates best 2-loop SUSY contribution to a_mu without tan(beta)
48  * resummation.
49  *
50  * This function re-defines the muon Yukawa coupling in terms of the
51  * tree-level relation with the muon pole mass, i.e. \f$y_\mu =
52  * \frac{\sqrt{2} m_\mu^\text{pole}}{v_d}\f$. Therefore, this
53  * function does not use tan(beta) resummation.
54  */
55 double calculate_amu_2loop_non_tan_beta_resummed(const MSSMNoFV_onshell& model)
56 {
57  MSSMNoFV_onshell model_ytree(model);
58  model_ytree.convert_to_non_tan_beta_resummed();
59 
60  return amu2LFSfapprox_non_tan_beta_resummed(model_ytree)
61  + amu2LChipmPhotonic(model_ytree)
62  + amu2LChi0Photonic(model_ytree)
63  + amu2LaSferm(model_ytree)
64  + amu2LaCha(model_ytree);
65 }

82 /**
83  * Calculates \f$m_{SUSY}\f$, p.37 arxiv:1311.1775.
84  * Finds minimum of special masses to normalize logarithms.
85  */
86 double log_scale(const MSSMNoFV_onshell& model)
87 {
88  return std::fmin(std::abs(model.get_MassB()),
89  std::fmin(std::abs(model.get_MassWB()),
90  std::fmin(std::abs(model.get_Mu()),
91  std::fmin(std::sqrt(model.get_me2(1, 1)),
92  std::sqrt(model.get_ml2(1, 1))))));
93 }

95 /**
96  * Calculates \f$\Delta_{g_1}\f$, Eq (6.6a) arxiv:1311.1775.
97  *
98  * Contributions from 1st and 2nd generation sleptons have been
99  * included in addition.
100  */
101 double delta_g1(const MSSMNoFV_onshell& model)
102 {
103  const double gY = model.get_gY();
104  const Eigen::Matrix<double,3,3>& mu2(model.get_mu2());
105  const Eigen::Matrix<double,3,3>& md2(model.get_md2());
106  const Eigen::Matrix<double,3,3>& mq2(model.get_mq2());
107  const Eigen::Matrix<double,3,3>& me2(model.get_me2());
108  const Eigen::Matrix<double,3,3>& ml2(model.get_ml2());
109  const double logscale = log_scale(model);
110 
111  return sqr(gY) * oneOver16PiSqr * 4. / 3. *
112  (4. / 3. * std::log(std::sqrt(mu2(0, 0)) / logscale) +
113  4. / 3. * std::log(std::sqrt(mu2(1, 1)) / logscale) +
114  4. / 3. * std::log(std::sqrt(mu2(2, 2)) / logscale) +
115  1. / 3. * std::log(std::sqrt(md2(0, 0)) / logscale) +
116  1. / 3. * std::log(std::sqrt(md2(1, 1)) / logscale) +
117  1. / 3. * std::log(std::sqrt(md2(2, 2)) / logscale) +
118  1. / 6. * std::log(std::sqrt(mq2(0, 0)) / logscale) +
119  1. / 6. * std::log(std::sqrt(mq2(1, 1)) / logscale) +
120  1. / 6. * std::log(std::sqrt(mq2(2, 2)) / logscale) +
121  std::log(std::sqrt(me2(0, 0)) / logscale) +
122  std::log(std::sqrt(me2(1, 1)) / logscale) +
123  std::log(std::sqrt(me2(2, 2)) / logscale) +
124  0.5 * std::log(std::sqrt(ml2(0, 0)) / logscale) +
125  0.5 * std::log(std::sqrt(ml2(1, 1)) / logscale) +
126  0.5 * std::log(std::sqrt(ml2(2, 2)) / logscale));
127 }

129 /**
130  * Calculates \f$\Delta_{\tilde{H}}\f$, Eq (6.6c) arxiv:1311.1775.
131  */
132 double delta_yuk_higgsino(const MSSMNoFV_onshell& model)
133 {
134  const double ytau = model.get_Ye(2, 2);
135  const double ytop = model.get_Yu(2, 2);
136  const double ybot = model.get_Yd(2, 2);
137  const Eigen::Matrix<double,3,3>& mu2(model.get_mu2());
138  const Eigen::Matrix<double,3,3>& md2(model.get_md2());
139  const Eigen::Matrix<double,3,3>& mq2(model.get_mq2());
140  const Eigen::Matrix<double,3,3>& me2(model.get_me2());
141  const Eigen::Matrix<double,3,3>& ml2(model.get_ml2());
142  const double logscale = log_scale(model);
143 
144  return oneOver16PiSqr * 0.5 *
145  (3. * sqr(ytop) * std::log(std::sqrt(mu2(2, 2)) / logscale) +
146  3. * sqr(ybot) * std::log(std::sqrt(md2(2, 2)) / logscale) +
147  3. * (sqr(ytop) + sqr(ybot)) *
148  std::log(std::sqrt(mq2(2, 2)) / logscale) +
149  sqr(ytau) * (std::log(std::sqrt(me2(2, 2)) / logscale) +
150  std::log(std::sqrt(ml2(2, 2)) / logscale)));
151 }

169 /**
170  * Calculates \f$\Delta_{g_2}\f$, Eq (6.6b) arxiv:1311.1775.
171  *
172  * Contributions from 1st and 2nd generation sleptons have been
173  * included in addition.
174  */
175 double delta_g2(const MSSMNoFV_onshell& model)
176 {
177  const double g2 = model.get_g2();
178  const Eigen::Matrix<double,3,3>& mq2(model.get_mq2());
179  const Eigen::Matrix<double,3,3>& ml2(model.get_ml2());
180  const double logscale = log_scale(model);
181 
182  return sqr(g2) * oneOver16PiSqr * 4. / 3. *
183  (1.5 * std::log(std::sqrt(mq2(0, 0)) / logscale) +
184  1.5 * std::log(std::sqrt(mq2(1, 1)) / logscale) +
185  1.5 * std::log(std::sqrt(mq2(2, 2)) / logscale) +
186  0.5 * std::log(std::sqrt(ml2(0, 0)) / logscale) +
187  0.5 * std::log(std::sqrt(ml2(1, 1)) / logscale) +
188  0.5 * std::log(std::sqrt(ml2(2, 2)) / logscale));
189 }

205 /**
206  * Calculates \f$\Delta_{t_\beta}\f$, Eq (6.6f) arxiv:1311.1775.

208 double delta_tan_beta(const MSSMNoFV_onshell& model)
209 {

211  const double ytop = model.get_Yu(2, 2);
212  const double ybot = model.get_Yd(2, 2);

295 /**
296  * Calculates the photonic 2-loop contribution to the 1-loop chargino
297  * diagram, Eq (35) arXiv:1003.5820.
298  */
299 double amu2LChipmPhotonic(const MSSMNoFV_onshell& model)
300 {
301  const double MM = model.get_MM();
302  const Eigen::Array<double,2,1> AAC_(AAC(model));
303  const Eigen::Array<double,2,1> BBC_(BBC(model));
304  const Eigen::Array<double,2,1>& MCha(model.get_MCha());
305  const double MSvmL = model.get_MSvmL();
306  const Eigen::Array<double,2,1> x(x_k(model));
307  const double Q = model.get_scale();
308 
309  double result = 0.;
310 
311  for (int k = 0; k < 2; k++) {
312  result += (AAC_(k) * F1C(x(k)) / 12.
313  + MCha(k) / 3. * BBC_(k) / model.get_MM() * F2C(x(k)))
314  * 16. * std::log(MM / MSvmL)
315  - 47. * AAC_(k) * F3C(x(k)) / 72.
316  - 61. * MCha(k) / 9. * BBC_(k) / model.get_MM() * F4C(x(k))
317  - (0.5 * AAC_(k) * F1C(x(k))
318  + MCha(k) * BBC_(k) / model.get_MM() * F2C(x(k)))
319  * std::log(sqr(MSvmL / Q));
320  }
321 
322  return sqr(model.get_EL0()) * sqr(oneOver16PiSqr) * sqr(MM / MSvmL) * result;
323 }

325 /**
326  * Calculates the photonic 2-loop contribution to the 1-loop
327  * neutralino diagram, Eq (36) arXiv:1003.5820.
328  */
329 double amu2LChi0Photonic(const MSSMNoFV_onshell& model)
330 {
331  const double MM = model.get_MM();
332  const Eigen::Matrix<double,4,2> AAN_(AAN(model));
333  const Eigen::Matrix<double,4,2> BBN_(BBN(model));
334  const Eigen::Array<double,4,1>& MNeu(model.get_MChi());
335  const Eigen::Array<double,2,1>& MSmu(model.get_MSm());
336  const Eigen::Matrix<double,4,2> x(x_im(model));
337  const double Q = model.get_scale();
338 
339  double result = 0.;
340 
341  for (int i = 0; i < 4; ++i) {
342  for (int m = 0; m < 2; ++m) {
343  result += 1. / sqr(MSmu(m))
344  * ((- 1. / 12. * AAN_(i, m) * F1N(x(i, m))
345  - MNeu(i) / 6. * BBN_(i, m) / model.get_MM() * F2N(x(i, m)))
346  * 16. * std::log(MM / MSmu(m))
347  + 35. / 72. * AAN_(i, m) * F3N(x(i, m))
348  + 8. * MNeu(i) / 9. * BBN_(i, m) / model.get_MM() * F4N(x(i, m))
349  + (0.25 * AAN_(i, m) * F1N(x(i, m)))
350  * std::log(sqr(MSmu(m) / Q)) );
351  }
352  }
353 
354  return sqr(model.get_EL0()) * sqr(oneOver16PiSqr) * sqr(MM) * result;
355 }

364 /**
365  * Calculates CP-even Higgs mixing angle \f$\tan\alpha\f$.
366  *
367  * @note the result is < 0
368  */
369 double tan_alpha(const MSSMNoFV_onshell& model)
370 {
371  const double TB = model.get_TB();
372  const double MZ = model.get_MZ();
373  const double MA0 = model.get_MA0();
374  const double tan2beta = 2. * TB / (1. - sqr(TB));
375  const double tan2alpha = tan2beta * (sqr(MA0) + sqr(MZ)) / (sqr(MA0) - sqr(MZ));
376 
377  return -1.0 / tan2alpha - std::sqrt(1.0 / sqr(tan2alpha) + 1.0);
378 }

380 /**
381  * Calculates \f$\lambda_{\mu}\f$, Eq (65), and
382  * \f$\lambda_{\chi_k^+}\f$, Eq (66) arXiv:hep-ph/0609168.
383  *
384  * Row 0 contains \f$\lambda_{\chi_1^+}\f$, Eq. (66).
385  * Row 1 contains \f$\lambda_{\chi_2^+}\f$, Eq. (66).
386  * Row 2 contains \f$\lambda_{\mu}\f$, Eq. (65).
387  *
388  * includes tan(beta) resummation
389  */
390 Eigen::Matrix<std::complex<double>,3,3> lambda_mu_cha(const MSSMNoFV_onshell& model)
391 {
392  const Eigen::Array<double,2,1>& MCha(model.get_MCha());
393  const double MW(model.get_MW());
394  const double TB(model.get_TB());
395  const double CB = 1. / std::sqrt(1. + sqr(TB));
396  const double SB = std::sqrt(1. - sqr(CB));
397  const double TA(tan_alpha(model));
398  const double CA = 1. / std::sqrt(1. + sqr(TA));
399  const double SA = - std::sqrt(1. - sqr(CA));
400  const Eigen::Matrix<std::complex<double>,2,2>& U(model.get_UM());
401  const Eigen::Matrix<std::complex<double>,2,2>& V(model.get_UP());
402  const double one_over_cb_eff = root2 * model.get_Ye(1,1)
403  * model.get_MW() / model.get_MM() / model.get_g2();
404 
405  Eigen::Matrix<std::complex<double>,3,3> result;
406 
407  for (int k = 0; k < 2; ++k) {
408  result(k, 0) = root2 * MW / MCha(k)
409  * (U(k, 0) * V(k, 1) * CA + U(k, 1) * V(k, 0) * (-SA));
410  result(k, 1) = root2 * MW / MCha(k)
411  * (U(k, 0) * V(k, 1) * SA + U(k, 1) * V(k, 0) * CA);
412  result(k, 2) = root2 * MW / MCha(k)
413  * (U(k, 0) * V(k, 1) * (-CB) + U(k, 1) * V(k, 0) * (-SB));
414  }
415  result(2, 0) = -SA * one_over_cb_eff;
416  result(2, 1) = CA * one_over_cb_eff;
417  result(2, 2) = SB * one_over_cb_eff;
418 
419  return result;
420 }

422 /**
423  * Calculates \f$\lambda_{\tilde{b}_i}\f$, Eq (67)
424  * arXiv:hep-ph/0609168
425  */
426 Eigen::Matrix<std::complex<double>,2,2> lambda_stop(const MSSMNoFV_onshell& model) {
427  const double TB(model.get_TB());
428  const double CB = 1. / std::sqrt(1. + sqr(TB));
429  const double SB = std::sqrt(1. - sqr(CB));
430  const double TA(tan_alpha(model));
431  const double CA = 1. / std::sqrt(1. + sqr(TA));
432  const double SA = - std::sqrt(1. - sqr(CA));
433  const double MT(model.get_MT());
434  const Eigen::Array<double,2,1>& MStop(model.get_MSt());
435  const Eigen::Matrix<double,2,2>& UStop(model.get_USt());
436  const double At(model.get_Au(2, 2));
437  const double Mu(model.get_Mu());
438 
439  Eigen::Matrix<std::complex<double>,2,2> result;
440 
441  for (int i = 0; i < 2; ++i) {
442  result(i, 0) = 2. * MT / (sqr(MStop(i)) * SB) * (Mu * SA + At * CA)
443  * std::conj(UStop(i, 0)) * UStop(i, 1);
444  result(i, 1) = 2. * MT / (sqr(MStop(i)) * SB) * (Mu * (-CA) + At * SA)
445  * std::conj(UStop(i, 0)) * UStop(i, 1);
446  }
447 
448  return result;
449 }

451 /**
452  * Calculates \f$\lambda_{\tilde{b}_i}\f$, Eq (68)
453  * arXiv:hep-ph/0609168
454  *
455  * includes tan(beta) resummation
456  */
457 Eigen::Matrix<std::complex<double>,2,2> lambda_sbot(const MSSMNoFV_onshell& model)
458 {
459  const double TA(tan_alpha(model));
460  const double CA = 1. / std::sqrt(1. + sqr(TA));
461  const double SA = - std::sqrt(1. - sqr(CA));
462  const Eigen::Array<double,2,1>& MSbot(model.get_MSb());
463  const Eigen::Matrix<double,2,2>& USbot(model.get_USb());
464  const double Ab(model.get_Ad(2, 2));
465  const double Mu(model.get_Mu());
466  const double mb_over_cb_eff = root2 * model.get_Yd(2,2)
467  * model.get_MW() / model.get_g2();
468 
469  Eigen::Matrix<std::complex<double>,2,2> result;
470 
471  for (int i = 0; i < 2; ++i) {
472  result(i, 0) = 2. * mb_over_cb_eff / (sqr(MSbot(i))) * (- Mu * CA + Ab * (-SA))
473  * std::conj(USbot(i, 0)) * USbot(i, 1);
474  result(i, 1) = 2. * mb_over_cb_eff / (sqr(MSbot(i))) * (- Mu * SA + Ab * CA)
475  * std::conj(USbot(i, 0)) * USbot(i, 1);
476  }
477 
478  return result;
479 }

481 /**
482  * Calculates \f$\lambda_{\tilde{t}_i}\f$, Eq (69)
483  * arXiv:hep-ph/0609168
484  *
485  * includes tan(beta) resummation
486  */
487 Eigen::Matrix<std::complex<double>,2,2> lambda_stau(const MSSMNoFV_onshell& model)
488 {
489  const double TA(tan_alpha(model));
490  const double CA = 1. / std::sqrt(1. + sqr(TA));
491  const double SA = - std::sqrt(1. - sqr(CA));
492  const Eigen::Array<double,2,1>& MStau(model.get_MStau());
493  const Eigen::Matrix<double,2,2>& UStau(model.get_UStau());
494  const double Al(model.get_Ae(2, 2));
495  const double Mu(model.get_Mu());
496  const double mtau_over_cb_eff = root2 * model.get_Ye()(2,2)
497  * model.get_MW() / model.get_g2();
498 
499  Eigen::Matrix<std::complex<double>,2,2> result;
500 
501  for (int i = 0; i < 2; ++i) {
502  result(i, 0) = 2. * mtau_over_cb_eff / (sqr(MStau(i))) * (- Mu * CA + Al * (-SA))
503  * std::conj(UStau(i, 0)) * UStau(i, 1);
504  result(i, 1) = 2. * mtau_over_cb_eff / (sqr(MStau(i))) * (- Mu * SA + Al * CA)
505  * std::conj(UStau(i, 0)) * UStau(i, 1);
506  }
507 
508  return result;
509 }

511 /**
512  * Calculates 2-loop contribution to amu, where a sfermion loop has
513  * been inserted into a 1-loop Standard Model diagram (photonic
514  * Barr-Zee diagram \f$(\tilde{f}\gamma H)\f$), Eq (64)
515  * arXiv:hep-ph/0609168.
516  */
517 double amu2LaSferm(const MSSMNoFV_onshell& model)
518 {
519  const double MM(model.get_MM());
520  const double MW(model.get_MW());
521  const double SW = std::sqrt(1. - sqr(MW / model.get_MZ()));
522  const double EL(model.get_EL());
523  const Eigen::Array<double,2,1>& m_stop(model.get_MSt());
524  const Eigen::Array<double,2,1>& m_sbot(model.get_MSb());
525  const Eigen::Array<double,2,1>& m_stau(model.get_MStau());
526  const Eigen::Array<double,2,1>& M_higgs(model.get_Mhh());
527  const Eigen::Matrix<std::complex<double>,3,3> lambda(lambda_mu_cha(model));
528  Eigen::Array<std::complex<double>,2,1> lambda_mu;
529  lambda_mu(0) = lambda(2, 0);
530  lambda_mu(1) = lambda(2, 1);
531  const Eigen::Matrix<std::complex<double>,2,2> lambdastop(lambda_stop(model));
532  const Eigen::Matrix<std::complex<double>,2,2> lambdasbot(lambda_sbot(model));
533  const Eigen::Matrix<std::complex<double>,2,2> lambdastau(lambda_stau(model));
534 
535  double result = 0.;
536  double N_c = 3.;
537  double Q = 2./3.;
538 
539  for (int i = 0; i < 2; ++i) {
540  for (int s = 0; s < 2; ++s) {
541  result += N_c * sqr(Q) * real(lambda_mu(s)
542  * lambdastop(i, s))
543  * f_sferm(sqr(m_stop(i) / M_higgs(s)));
544  }
545  }
546 
547  Q = -1./3.;
548  for (int i = 0; i < 2; ++i) {
549  for (int s = 0; s < 2; ++s) {
550  result += N_c * sqr(Q) * real(lambda_mu(s)
551  * lambdasbot(i, s))
552  * f_sferm(sqr(m_sbot(i) / M_higgs(s)));
553  }
554  }
555 
556  N_c = 1.;
557  Q = -1.;
558  for (int i = 0; i < 2; ++i) {
559  for (int s = 0; s < 2; ++s) {
560  result += N_c * sqr(Q) * real(lambda_mu(s)
561  * lambdastau(i, s))
562  * f_sferm(sqr(m_stau(i) / M_higgs(s)));
563  }
564  }
565 
566  return result * 2. * sqr(oneOver16PiSqr * sqr(EL) * MM / (MW * SW));
567 }
568 
569 /**
570  * Calculates 2-loop contribution to amu, where a chargino loop has
571  * been inserted into a 1-loop Standard Model diagram (photonic
572  * Barr-Zee diagram \f$(\chi\gamma H)\f$), Eq (63)
573  * arXiv:hep-ph/0609168.
574  */
575 double amu2LaCha(const MSSMNoFV_onshell& model)
576 {
577  const double MM(model.get_MM());
578  const double MW(model.get_MW());
579  const double MA0(model.get_MA0());
580  const Eigen::Array<double,2,1>& M_higgs(model.get_Mhh());
581  const double SW = std::sqrt(1. - sqr(MW / model.get_MZ()));
582  const double EL(model.get_EL());
583  const Eigen::Array<double,2,1>& m_cha(model.get_MCha());
584  const Eigen::Matrix<std::complex<double>,3,3> lambda(lambda_mu_cha(model));
585 
586  double result = 0.;
587 
588  for (int k = 0; k < 2; ++k) {
589  result += real(lambda(2, 2) * lambda(k, 2)) * f_PS(sqr(m_cha(k) / MA0));
590  for (int s = 0; s < 2; ++s) {
591  result += real(lambda(2, s) * lambda(k, s))
592  * f_S(sqr(m_cha(k) / M_higgs(s)));
593  }
594  }
595 
596  return result * 2. * sqr(oneOver16PiSqr * sqr(EL) * MM / (MW * SW));
597 }
