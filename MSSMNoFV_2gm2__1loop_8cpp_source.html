48 /**
49  * Calculates full 1-loop SUSY contribution to (g-2), Eq (45) of
50  * arXiv:hep-ph/0609168.
51  *
52  * This function re-defines the muon Yukawa coupling in terms of the
53  * tree-level relation with the muon pole mass, i.e. \f$y_\mu =
54  * \frac{\sqrt{2} m_\mu^\text{pole}}{v_d}\f$. Therefore, this
55  * function does not use tan(beta) resummation.
56  */
57 double calculate_amu_1loop_non_tan_beta_resummed(const MSSMNoFV_onshell& model)
58 {
59  // create new model and reset its muon Yukawa coupling to tree-level
60  MSSMNoFV_onshell model_ytree(model);
61  model_ytree.convert_to_non_tan_beta_resummed();
62 
63  return amu1LChi0(model_ytree) + amu1LChipm(model_ytree);
64 }

66 /**
67  * Calculates full 1-loop SUSY contribution to (g-2), Eq (45) of
68  * arXiv:hep-ph/0609168.
69  *
70  * This function assumes that the Yukawa coupling is defined according
71  * to arXiv:1504.05500, Eq. (13) and footnote 2. Therefore, this
72  * function uses tan(beta) resummation in the Yukawa coupling.
73  */
74 double calculate_amu_1loop(const MSSMNoFV_onshell& model)
75 {
76  return amu1LChi0(model) + amu1LChipm(model);

79 /**
80  * Calculates 1-loop neutralino contribution to (g-2), Eq (2.11a) of
81  * arXiv:1311.1775.
82  */
83 double amu1LChi0(const MSSMNoFV_onshell& model)
84 {
85  const Eigen::Array<double,2,1>& m_smu(model.get_MSm());
86  const Eigen::Matrix<double,4,2> AAN_(AAN(model));
87  const Eigen::Matrix<double,4,2> BBN_(BBN(model));
88  const Eigen::Matrix<double,4,2> x(x_im(model));
89  const Eigen::Array<double,4,1>& MChi(model.get_MChi());
90 
91  double result = 0.;
92 
93  for (int i = 0; i < 4; ++i) {
94  for (int m = 0; m < 2; ++m) {
95  result += - AAN_(i, m) * F1N(x(i, m)) / (12. * sqr(m_smu(m)))
96  - MChi(i) * BBN_(i, m) * F2N(x(i, m))
97  / (6. * model.get_MM() * sqr(m_smu(m)));
98  }
99  }
100 
101  return result * sqr(model.get_MM()) * oneOver16PiSqr;
102 }

104 /**
105  * Calculates 1-loop chargino contribution to (g-2), Eq (2.11b) of
106  * arXiv:1311.1775.
107  */
108 double amu1LChipm(const MSSMNoFV_onshell& model)
109 {
110  const Eigen::Array<double,2,1> x(x_k(model));
111  const double MSvm(model.get_MSvmL());
112  const Eigen::Array<double,2,1> AAC_(AAC(model));
113  const Eigen::Array<double,2,1> BBC_(BBC(model));
114  const Eigen::Array<double,2,1>& MCha(model.get_MCha());
115 
116  double result = 0.;
117 
118  for (int k = 0; k < 2; ++k) {
119  result +=
120  AAC_(k) * F1C(x(k)) / 12 +
121  MCha(k) * BBC_(k) * F2C(x(k)) / (3 * model.get_MM());
122  }
123 
124  return result * sqr(model.get_MM()/MSvm) * oneOver16PiSqr;
125 }

127 /**
128  * Calculates \f$n^L_{i\tilde{l}_k}\f$, Eq (2.5o) of arXiv:1311.1775,
129  * for \f$l=\mu\f$
130  */
131 Eigen::Matrix<std::complex<double>,4,2> n_L(const MSSMNoFV_onshell& model)
132 {
133  const double gY(model.get_gY());
134  const double g2(model.get_g2());
135  const double ymu(model.get_Ye(1, 1));
136  const Eigen::Matrix<std::complex<double>,4,4>& ZN(model.get_ZN());
137  const Eigen::Matrix<double,2,2>& u_smu(model.get_USm());
138 
139  Eigen::Matrix<std::complex<double>,4,2> result;
140 
141  for (int i = 0; i < 4; ++i) {
142  for (int m = 0; m < 2; ++m) {
143  result(i, m) =
144  1. / root2 * (gY * std::conj(ZN(i,0))
145  + g2 * std::conj(ZN(i,1))) * u_smu(m,0)
146  - ymu * std::conj(ZN(i,2)) * u_smu(m,1);
147  }
148  }
149 
150  return result;
151 }

153 /**
154  * Calculates \f$n^R_{i\tilde{l}_k}\f$, Eq (2.5p) of arXiv:1311.1775,
155  * with \f$l=\mu\f$.
156  */
157 Eigen::Matrix<std::complex<double>,4,2> n_R(const MSSMNoFV_onshell& model)
158 {
159  const double gY(model.get_gY());
160  const double ymu(model.get_Ye(1, 1));
161  const Eigen::Matrix<std::complex<double>,4,4>& ZN(model.get_ZN());
162  const Eigen::Matrix<double,2,2>& u_smu(model.get_USm());
163 
164  Eigen::Matrix<std::complex<double>,4,2> result;
165 
166  for (int i = 0; i < 4; ++i) {
167  for (int m = 0; m < 2; ++m) {
168  result(i, m) = -(root2 * gY * ZN(i, 0) * u_smu(m, 1) +
169  ymu * ZN(i, 2) * u_smu(m, 0));
170  }
171  }
172 
173  return result;
174 }

176 /**
177  * Calculates \f$c^L_{i\tilde{\nu}_\mu}\f$, Eq (2.5a) of
178  * arXiv:1311.1775 for \f$l=\mu\f$.
179  *
180  * This expression is the complex conjugate of Eq. (50) of
181  * arXiv:hep-ph/0609168.
182  */
183 Eigen::Array<std::complex<double>,2,1> c_L(const MSSMNoFV_onshell& model)
184 {
185  const double g2 = model.get_g2();
186  const Eigen::Matrix<std::complex<double>,2,2>& UP(model.get_UP());
187 
188  Eigen::Array<std::complex<double>,2,1> result;
189 
190  for (int k = 0; k < 2; ++k) {
191  result(k) = std::conj(UP(k, 0));
192  }
193 
194  return -g2 * result;
195 }

198  * Calculates \f$c^R_{i\tilde{\nu}_\mu}\f$, Eq (2.5b) of
199  * arXiv:1311.1775 for \f$l=\mu\f$.

201  * This expression is equal to Eq. (51) of arXiv:hep-ph/0609168.
202  */
203 Eigen::Array<std::complex<double>,2,1> c_R(const MSSMNoFV_onshell& model)
204 {

211  result(k) = UM(k, 1);
212  }
213

217 /**
218  * Calculates \f$\mathcal{A}^{c+}_{ii\tilde{\nu}_\mu} =
219  * |c^L_{i\tilde{\nu}_\mu}|^2 + |c^R_{i\tilde{\nu}_\mu}|^2\f$,
220  * Eqs (2.11b), (2.7a) in arXiv:1311.1775.
221  *
222  * This expression is identical to \f$|c^L_{im}|^2 + |c^R_{im}|^2\f$,
223  * which appears in Eq. (47) of arXiv:hep-ph/0609168.
224  */
225 Eigen::Array<double,2,1> AAC(const MSSMNoFV_onshell& model)
226 {
227  const Eigen::Array<std::complex<double>,2,1> cL(c_L(model));
228  const Eigen::Array<std::complex<double>,2,1> cR(c_R(model));
229 
230  Eigen::Array<double,2,1> result;
231 
232  for (int k = 0; k < 2; ++k) {
233  result(k) = norm(cL(k)) + norm(cR(k));
234  }
235 
236  return result;
237 }

239 /**
240  * Calculates \f$\mathcal{A}^{n+}_{ii\tilde{\mu}_m} =
241  * |n^L_{i\tilde{\mu}_m}|^2 + |n^R_{i\tilde{\mu}_m}|^2\f$,
242  * Eqs (2.11a), (2.7a) in arXiv:1311.1775.
243  *
244  * This expression is identical to \f$|n^L_{im}|^2 + |n^R_{im}|^2\f$,
245  * which appears in Eq. (46) of arXiv:hep-ph/0609168.
246  */
247 Eigen::Matrix<double,4,2> AAN(const MSSMNoFV_onshell& model)
248 {

250  const Eigen::Matrix<std::complex<double>,4,2> nR(n_R(model));
251

253 
254  for (int i = 0; i < 4; ++i) {
255  for (int m = 0; m < 2; ++m) {
256  result(i, m) = norm(nL(i, m)) + norm(nR(i, m));
257  }
258  }
259 
260  return result;

263 /**
264  * Calculates \f$\mathcal{B}^{c+}_{ii\tilde{\nu}_\mu} = 2 \text{Re}
265  * [(c^L_{i\tilde{\nu}_\mu})^* c^R_{i\tilde{\nu}_\mu}]\f$,
266  * Eqs (2.11b), (2.7b) in arXiv:1311.1775.
267  */
268 Eigen::Array<double,2,1> BBC(const MSSMNoFV_onshell& model)

273  Eigen::Array<double,2,1> result;

275  for (int k = 0; k < 2; ++k) {
276  result(k) = 2. * std::real(std::conj(cL(k)) * cR(k));

282 /**
283  * Calculates \f$\mathcal{B}^{n+}_{ii\tilde{\mu}_m} = 2 \text{Re}
284  * [(n^L_{i\tilde{\mu}_m})^* n^R_{i\tilde{\mu}_m}]\f$, Eqs (2.11a),
285  * (2.7b) in arXiv:1311.1775.
286  */
287 Eigen::Matrix<double,4,2> BBN(const MSSMNoFV_onshell& model)
288 {

303 /**

339 /**
340  * Calculates the 1-loop leading log approximation: Wino--Higgsino,
341  * muon-sneutrino, Eq (6.2a) arXiv:1311.1775
342  */
343 double amu1LWHnu(const MSSMNoFV_onshell& model)
344 {
345  const double tan_beta = model.get_TB();
346  const double M2 = model.get_MassWB();
347  const double MUE = model.get_Mu();
348  const double MSv_2 = model.get_MSvmL();
349 
350  return sqr(model.get_g2()) * 2. * oneOver16PiSqr
351  * (sqr(model.get_MM()) * M2 * MUE * tan_beta) / sqr(sqr(MSv_2))
352  * Fa(sqr(M2 / MSv_2), sqr(MUE / MSv_2));
353 }

355 /**
356  * Calculates the 1-loop leading log approximation: Wino--Higgsino,
357  * left-handed smuon, Eq (6.2b) arXiv:1311.1775
358  */
359 double amu1LWHmuL(const MSSMNoFV_onshell& model)
360 {
361  const double tan_beta = model.get_TB();
362  const double M2 = model.get_MassWB();
363  const double MUE = model.get_Mu();
364  const double MSL_2 = std::sqrt(model.get_ml2(1, 1));
365 
366  return - sqr(model.get_g2()) * oneOver16PiSqr
367  * (sqr(model.get_MM()) * M2 * MUE * tan_beta) / sqr(sqr(MSL_2))
368  * Fb(sqr(M2 / MSL_2), sqr(MUE / MSL_2));
369 }

371 /**
372  * Calculates the 1-loop leading log approximation: Bino--Higgsino,
373  * left-handed smuon, Eq (6.2c) arXiv:1311.1775
374  */
375 double amu1LBHmuL(const MSSMNoFV_onshell& model)
376 {
377  const double tan_beta = model.get_TB();
378  const double M1 = model.get_MassB();
379  const double MUE = model.get_Mu();
380  const double MSL_2 = std::sqrt(model.get_ml2(1, 1));
381  const double gY = model.get_gY();
382 
383  return sqr(gY) * oneOver16PiSqr
384  * (sqr(model.get_MM()) * M1 * MUE * tan_beta) / sqr(sqr(MSL_2))
385  * Fb(sqr(M1 / MSL_2), sqr(MUE / MSL_2));
386 }

388 /**
389  * Calculates the 1-loop leading log approximation: Bino--Higgsino,
390  * right-handed smuon, Eq (6.2d) arXiv:1311.1775
391  */
392 double amu1LBHmuR(const MSSMNoFV_onshell& model)
393 {
394  const double tan_beta = model.get_TB();
395  const double M1 = model.get_MassB();
396  const double MUE = model.get_Mu();
397  const double MSE_2 = std::sqrt(model.get_me2(1, 1));
398  const double gY = model.get_gY();
399 
400  return - sqr(gY) * 2. * oneOver16PiSqr
401  * (sqr(model.get_MM()) * M1 * MUE * tan_beta) / sqr(sqr(MSE_2))
402  * Fb(sqr(M1 / MSE_2), sqr(MUE / MSE_2));
403 }

405 /**
406  * Calculates the 1-loop leading log approximation: Bino, left-handed
407  * smuon, right-handed smuon, Eq (6.2e) arXiv:1311.1775
408  */
409 double amu1LBmuLmuR(const MSSMNoFV_onshell& model)
410 {
411  const double tan_beta = model.get_TB();
412  const double M1 = model.get_MassB();
413  const double MUE = model.get_Mu();
414  const double MSL_2 = std::sqrt(model.get_ml2(1, 1));
415  const double MSE_2 = std::sqrt(model.get_me2(1, 1));
416  const double gY = model.get_gY();
417 
418  if (is_zero(M1, eps)) {
419  return 0.;
420  }
421 
422  return sqr(gY) * 2. * oneOver16PiSqr
423  * (sqr(model.get_MM()) * MUE * tan_beta) / (M1 * sqr(M1))
424  * Fb(sqr(MSL_2 / M1), sqr(MSE_2 / M1));
425 }

464 /**
465  * Calculates \f$\Delta_{\ell}\f$ using a generalized form of Eq (8)
466  * arxiv:0808.1530.
467  *
468  * @param model model parameters
469  * @param gen lepton generation (0 = electron, 1 = muon, 2 = tau)
470  *
471  * @return \f$\Delta_{\ell}\f$
472  */
473 double delta_down_lepton_correction(const MSSMNoFV_onshell& model, int gen)
474 {
475  const double mu = model.get_Mu();
476  const double TB = model.get_TB();
477  const double g2 = model.get_g2();
478  const double gY = model.get_gY();
479  const double M1 = model.get_MassB();
480  const double M2 = model.get_MassWB();
481  const double MW = model.get_MW();
482  const double MZ = model.get_MZ();
483  const double SW = std::sqrt(1. - sqr(MW / MZ));
484 
485  const double m1 =
486  abs_sqrt(0.5 * (sqr(M2) + sqr(mu) + 2. * sqr(MW)
487  - abs_sqrt(sqr(sqr(M2) + sqr(mu) + 2. * sqr(MW))
488  - sqr(2. * M2 * mu))));
489  const double m2 =
490  abs_sqrt(0.5 * (sqr(M2) + sqr(mu) + 2. * sqr(MW)
491  + abs_sqrt(sqr(sqr(M2) + sqr(mu) + 2. * sqr(MW))
492  - sqr(2. * M2 * mu))));
493  const double m_sneu_lep = abs_sqrt(model.get_ml2(gen, gen) - 0.5 * sqr(MZ));
494  const double m_slep_L = abs_sqrt(model.get_ml2(gen, gen) - sqr(MZ) * (sqr(SW) - 0.5));
495  const double m_slep_R = abs_sqrt(model.get_me2(gen, gen) + sqr(MZ * SW));
496 
497  const double delta_lep =
498  - mu * TB * oneOver16PiSqr
499  * (sqr(g2) * M2 * (Iabc(m1, m2, m_sneu_lep) + 0.5 * Iabc(m1, m2, m_slep_L))
500  + sqr(gY) * M1 * (Iabc(mu, M1, m_slep_R) - 0.5 * Iabc(mu, M1, m_slep_L)
501  - Iabc(M1, m_slep_L, m_slep_R)));
502 
503  return delta_lep;
504 }

530 /**
531  * Returns the \f$\Delta_b\f$ corrections from arxiv:0901.2065,
532  * Eq (103) and Eqs (31)-(35)
533  */
534 double delta_bottom_correction(const MSSMNoFV_onshell& model)
535 {
536  const double TB = model.get_TB();
537  const double At = model.get_Au(2,2);
538  const double yt = model.get_Yu(2,2);
539  const double gY = model.get_gY();
540  const double g2 = model.get_g2();
541  const double alpha_S = sqr(model.get_g3()) / (4*Pi);
542  const double mu = model.get_Mu();
543  const double M1 = model.get_MassB();
544  const double M2 = model.get_MassWB();
545  const double M3 = model.get_MassG();
546  const double mstL2 = std::abs(model.get_mq2(2,2));
547  const double mstR2 = std::abs(model.get_mu2(2,2));
548  const double msbL2 = std::abs(model.get_mq2(2,2));
549  const double msbR2 = std::abs(model.get_md2(2,2));
550 
551  // Note:
552  // 1/z^2 H_2(x^2/z^2, y^2/z^2) = - Iabc(x,y,z)
553 
554  // Eq.(31)
555  const double eps_0 =
556  2. * alpha_S / (3*Pi) * mu * M3
557  * Iabc(abs_sqrt(msbL2), abs_sqrt(msbR2), std::abs(M3))
558 
559  - sqr(gY) / (96.*sqr(Pi)) * mu * M1
560  * (Iabc(abs_sqrt(msbL2), std::abs(mu), std::abs(M1))
561  + 2. * Iabc(abs_sqrt(msbR2), std::abs(mu), std::abs(M1)))
562 
563  - sqr(gY) / (144.*sqr(Pi)) * mu * M1
564  * Iabc(abs_sqrt(msbL2), abs_sqrt(msbR2), std::abs(M1))
565 
566  - 3. * sqr(g2) / (32.*sqr(Pi)) * mu * M2
567  * Iabc(abs_sqrt(msbL2), std::abs(mu), std::abs(M2))
568  ;
569 
570  const double Ittm = Iabc(abs_sqrt(mstL2), abs_sqrt(mstR2), std::abs(mu));
571 
572  // Eq.(35), term ~ self-energy neglected
573  const double eps_Y_vM = - oneOver16PiSqr * At * mu * Ittm;
574 
575  // Eq.(32)
576  const double eps_Y = oneOver16PiSqr * At * mu * Ittm + eps_Y_vM;
577 
578  // Eq.(33)
579  const double eps_3_tilde = eps_0 + sqr(yt) * eps_Y;
580 
581  // Eq.(103)
582  const double delta_b = TB * eps_3_tilde;
583 
584  return delta_b;
