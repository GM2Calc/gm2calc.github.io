27template <typename T>
28struct Complex {
29 constexpr Complex(T re_ = T{}, T im_ = T{}) : re(re_), im(im_) {}
30 operator std::complex<T>() const noexcept { return std::complex<T>(re, im); }
31 T re{};
32 T im{};
33};

35template <typename T>
36constexpr T arg(const Complex<T>& z) noexcept
37{
38 return std::atan2(z.im, z.re);

41template <typename T>

47template <typename T>
48Complex<T> log(const Complex<T>& z) noexcept
49{
50 T a = arg(z);
51
52 if (z.im == T(0) && a < T(0)) {
53 a = -a;
54 }
55
56 return { 0.5*std::log(norm_sqr(z)), a };
57}

60constexpr T norm_sqr(const Complex<T>& z) noexcept
61{
