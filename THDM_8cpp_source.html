139 /**
140  * Initialize Yukawa couplings
141  */
142 void THDM::init_yukawas()
143 {
144  const Eigen::Matrix<double,3,3> mu = sm.get_mu().asDiagonal();
145  const Eigen::Matrix<double,3,3> md = sm.get_md().asDiagonal();
146  const Eigen::Matrix<double,3,3> ml = sm.get_ml().asDiagonal();
147  const Eigen::Matrix<std::complex<double>,3,3> vckm_adj = sm.get_ckm().adjoint();
148 
149  switch (yukawa_type) {
150  case thdm::Yukawa_type::type_1:
151  Pi_u.setZero();
152  Pi_d.setZero();
153  Pi_l.setZero();
154  set_Gamma_u(sqrt2*vckm_adj*mu/v1);
155  set_Gamma_d(sqrt2*md/v1);
156  set_Gamma_l(sqrt2*ml/v1);
157  break;
158  case thdm::Yukawa_type::type_2:
159  Gamma_u.setZero();
160  Pi_d.setZero();
161  Pi_l.setZero();
162  set_Pi_u(sqrt2*vckm_adj*mu/v2);
163  set_Gamma_d(sqrt2*md/v1);
164  set_Gamma_l(sqrt2*ml/v1);
165  break;
166  case thdm::Yukawa_type::type_X:
167  Gamma_u.setZero();
168  Gamma_d.setZero();
169  Pi_l.setZero();
170  set_Pi_u(sqrt2*vckm_adj*mu/v2);
171  set_Pi_d(sqrt2*md/v2);
172  set_Gamma_l(sqrt2*ml/v1);
173  break;
174  case thdm::Yukawa_type::type_Y:
175  Pi_u.setZero();
176  Pi_d.setZero();
177  Gamma_l.setZero();
178  set_Gamma_u(sqrt2*vckm_adj*mu/v1);
179  set_Gamma_d(sqrt2*md/v1);
180  set_Pi_l(sqrt2*ml/v2);
181  break;
182  case thdm::Yukawa_type::aligned:
183  set_Gamma_u(sqrt2*vckm_adj*mu/(v1 + v2*calc_xi(get_zeta_u(), get_tan_beta())));
184  set_Gamma_d(sqrt2*md/(v1 + v2*calc_xi(get_zeta_d(), get_tan_beta())));
185  set_Gamma_l(sqrt2*ml/(v1 + v2*calc_xi(get_zeta_l(), get_tan_beta())));
186  set_Pi_u(calc_xi(get_zeta_u(), get_tan_beta())*Gamma_u);
187  set_Pi_d(calc_xi(get_zeta_d(), get_tan_beta())*Gamma_d);
188  set_Pi_l(calc_xi(get_zeta_l(), get_tan_beta())*Gamma_l);
189  break;
190  case thdm::Yukawa_type::general:
191  set_Gamma_u(sqrt2*vckm_adj*mu/v1 - v2/v1*Pi_u);
192  set_Gamma_d(sqrt2*md/v1 - v2/v1*Pi_d);
193  set_Gamma_l(sqrt2*ml/v1 - v2/v1*Pi_l);
194  break;
195  }
196 }

198 /// Table 1, arxiv:1607.06292
199 double THDM::get_zeta_u() const
200 {
201  switch (yukawa_type) {
202  case thdm::Yukawa_type::type_1:
203  case thdm::Yukawa_type::type_2:
204  case thdm::Yukawa_type::type_X:
205  case thdm::Yukawa_type::type_Y:

208  return zeta_u;

210  return 0; // undefined in the general THDM

255 Eigen::Matrix<std::complex<double>,3,3> THDM::get_rho_u(const Eigen::Matrix<double,3,3>& mu) const
256 {
257  const double cb = get_cos_beta();
258  const double v = get_v();
259 
260  if (yukawa_type != thdm::Yukawa_type::general) {
261  return sqrt2*mu*get_zeta_u()/v + Delta_u;
262  }
263 
264  return get_Pi_u().real()/cb - sqrt2*mu*get_tan_beta()/v;
265 }

426 void THDM::set_basis(const thdm::Mass_basis& basis)
427 {
428  if (basis.mh > basis.mH) {
429  const char* err = "mh must be less than or equal to mH.";
430  if (config.force_output) {
431  WARNING(err);
432  } else {
433  throw EInvalidInput(err);
434  }
435  }
436 
437  if (basis.tan_beta <= 0) {
438  const char* err = "tan(beta) must be greater than zero.";
439  if (config.force_output) {
440  WARNING(err);
441  } else {
442  throw EInvalidInput(err);
443  }
444  }
445 
446  if (std::abs(basis.sin_beta_minus_alpha) > 1) {
447  const char* err = "|sin(beta - alpha_h)| must be less than or equal to 1.";
448  if (config.force_output) {
449  WARNING(err);
450  } else {
451  throw EInvalidInput(err);
452  }
453  }
454 
455  if (basis.mh < 0) {
456  const char* err = "mh must be greater than or equal to zero.";
457  if (config.force_output) {
458  WARNING(err);
459  } else {
460  throw EInvalidInput(err);
461  }
462  }
463 
464  if (basis.mH < 0) {
465  const char* err = "mH must be greater than or equal to zero.";
466  if (config.force_output) {
467  WARNING(err);
468  } else {
469  throw EInvalidInput(err);
470  }
471  }
472 
473  if (basis.mA < 0) {
474  const char* err = "mA must be greater than or equal to zero.";
475  if (config.force_output) {
476  WARNING(err);
477  } else {
478  throw EInvalidInput(err);
479  }
480  }
481 
482  if (basis.mHp < 0) {
483  const char* err = "mHp must be greater than or equal to zero.";
484  if (config.force_output) {
485  WARNING(err);
486  } else {
487  throw EInvalidInput(err);
488  }
489  }
490 
491  const double sba = basis.sin_beta_minus_alpha;
492  const double tb = basis.tan_beta;
493  const double ctb = 1./tb;
494  const double sb = tb/std::sqrt(1 + sqr(tb));
495  const double cb = 1./std::sqrt(1 + sqr(tb));
496  const double sb2 = sqr(sb);
497  const double cb2 = sqr(cb);
498  const double alpha = -std::asin(sba) + std::atan(tb);
499  const double sa = std::sin(alpha);
500  const double ca = std::cos(alpha);
501  const double sa2 = sqr(sa);
502  const double ca2 = sqr(ca);
503  const double mh = basis.mh;
504  const double mH = basis.mH;
505  const double mA = basis.mA;
506  const double mHp = basis.mHp;
507  const double lambda6 = basis.lambda_6;
508  const double lambda7 = basis.lambda_7;
509  const double m12_2 = basis.m122;
510  const double v = sm.get_v();
511  const double v2 = sqr(v);
512 
513  set_lambda1((sqr(mH)*ca2 + sqr(mh)*sa2 - m12_2*tb)/(v2*cb2) - 1.5*lambda6*tb + 0.5*lambda7*tb*tb*tb);
514  set_lambda2((sqr(mH)*sa2 + sqr(mh)*ca2 - m12_2*ctb)/(v2*sb2) + 0.5*lambda6*ctb*ctb*ctb - 1.5*lambda7*ctb);
515  set_lambda3(((sqr(mH) - sqr(mh))*ca*sa + 2.*sqr(mHp)*sb*cb - m12_2)/(v2*sb*cb) - 0.5*lambda6*ctb - 0.5*lambda7*tb);
516  set_lambda4(((sqr(mA) - 2.*sqr(mHp))*cb*sb + m12_2)/(v2*sb*cb) - 0.5*lambda6*ctb - 0.5*lambda7*tb);
517  set_lambda5((m12_2 - sqr(mA)*sb*cb)/v2/sb/cb - 0.5*lambda6*ctb - 0.5*lambda7*tb);
518  set_lambda6(basis.lambda_6);
519  set_lambda7(basis.lambda_7);
520  set_tan_beta_and_v(basis.tan_beta, v);
521  set_m122(basis.m122);
522  set_Pi_u(basis.Pi_u);
523  set_Pi_d(basis.Pi_d);
524  set_Pi_l(basis.Pi_l);
525 
526  validate();
527 
528  solve_ewsb();
529  init_yukawas();
530  calculate_MSbar_masses();
531 
532  if (get_problems().have_problem()) {
533  if (config.force_output) {
534  WARNING(get_problems().get_problems());
535  } else {
536  throw EPhysicalProblem(get_problems().get_problems());
537  }
538  }

598 void THDM::validate() const
599 {
600  // check zeta_f: only used in the flavour-aligned THDM
601  if (yukawa_type == thdm::Yukawa_type::type_1 ||
602  yukawa_type == thdm::Yukawa_type::type_2 ||
603  yukawa_type == thdm::Yukawa_type::type_X ||
604  yukawa_type == thdm::Yukawa_type::type_Y ||
605  yukawa_type == thdm::Yukawa_type::general) {
606  if (zeta_u != 0) {
607  WARNING("Value of zeta_u = " << zeta_u << " ignored, because Yukawa type is " << yukawa_type_to_string());
608  }
609  if (zeta_d != 0) {
610  WARNING("Value of zeta_d = " << zeta_d << " ignored, because Yukawa type is " << yukawa_type_to_string());
611  }
612  if (zeta_l != 0) {
613  WARNING("Value of zeta_l = " << zeta_l << " ignored, because Yukawa type is " << yukawa_type_to_string());
614  }
615  }
616 
617  // check Delta_f: not used in the general THDM
618  if (yukawa_type == thdm::Yukawa_type::general) {
619  if (Delta_u.cwiseAbs().maxCoeff() != 0) {
620  WARNING("Value of Delta_u ignored, because Yukawa type is " << yukawa_type_to_string());
621  }
622  if (Delta_d.cwiseAbs().maxCoeff() != 0) {
623  WARNING("Value of Delta_d ignored, because Yukawa type is " << yukawa_type_to_string());
624  }
625  if (Delta_l.cwiseAbs().maxCoeff() != 0) {
626  WARNING("Value of Delta_l ignored, because Yukawa type is " << yukawa_type_to_string());
627  }
628  }
629 
630  // check Pi_f: only used in the general THDM
631  if (yukawa_type == thdm::Yukawa_type::type_1 ||
632  yukawa_type == thdm::Yukawa_type::type_2 ||
633  yukawa_type == thdm::Yukawa_type::type_X ||
634  yukawa_type == thdm::Yukawa_type::type_Y ||
635  yukawa_type == thdm::Yukawa_type::aligned) {
636  if (get_Pi_u().cwiseAbs().maxCoeff() != 0) {
637  WARNING("Value of Pi_u ignored, because Yukawa type is " << yukawa_type_to_string());
638  }
639  if (get_Pi_d().cwiseAbs().maxCoeff() != 0) {
640  WARNING("Value of Pi_d ignored, because Yukawa type is " << yukawa_type_to_string());
641  }
642  if (get_Pi_l().cwiseAbs().maxCoeff() != 0) {
643  WARNING("Value of Pi_l ignored, because Yukawa type is " << yukawa_type_to_string());
644  }
