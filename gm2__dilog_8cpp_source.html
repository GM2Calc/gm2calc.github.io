28  template <int Nstart, typename T, int N>
29  Complex<T> horner(const Complex<T>& z, const T (&coeffs)[N]) noexcept
30  {
31  static_assert(0 <= Nstart && Nstart < N && N >= 2, "invalid array bounds");
32 
33  const T r = z.re + z.re;
34  const T s = z.re * z.re + z.im * z.im;
35  T a = coeffs[N - 1], b = coeffs[N - 2];
36 
37  for (int i = N - 3; i >= Nstart; --i) {
38  const T t = a;
39  a = b + r * a;
40  b = coeffs[i] - s * t;
41  }
42 
43  return Complex<T>(z.re*a + b, z.im*a);
44  }

48 /**
49  * @brief Real dilogarithm \f$\operatorname{Li}_2(x)\f$
50  * @param x real argument
51  * @return \f$\operatorname{Li}_2(x)\f$
52  * @author Alexander Voigt
53  *
54  * Implemented as an economized Pade approximation with a
55  * maximum error of 4.16e-18.
56  */
57 double dilog(double x) noexcept
58 {
59  const double PI = 3.1415926535897932;
60  const double P[] = {
61  0.9999999999999999502e+0,
62  -2.6883926818565423430e+0,
63  2.6477222699473109692e+0,
64  -1.1538559607887416355e+0,
65  2.0886077795020607837e-1,
66  -1.0859777134152463084e-2
67  };
68  const double Q[] = {
69  1.0000000000000000000e+0,
70  -2.9383926818565635485e+0,
71  3.2712093293018635389e+0,
72  -1.7076702173954289421e+0,
73  4.1596017228400603836e-1,
74  -3.9801343754084482956e-2,
75  8.2743668974466659035e-4
76  };
77 
78  double y = 0, r = 0, s = 1;
79 
80  // transform to [0, 1/2]
81  if (x < -1) {
82  const double l = std::log(1 - x);
83  y = 1/(1 - x);
84  r = -PI*PI/6 + l*(0.5*l - std::log(-x));
85  s = 1;
86  } else if (x == -1) {
87  return -PI*PI/12;
88  } else if (x < 0) {
89  const double l = std::log1p(-x);
90  y = x/(x - 1);
91  r = -0.5*l*l;
92  s = -1;
93  } else if (x == 0) {
94  return 0;
95  } else if (x < 0.5) {
96  y = x;
97  r = 0;
98  s = 1;
99  } else if (x < 1) {
100  y = 1 - x;
101  r = PI*PI/6 - std::log(x)*std::log(y);
102  s = -1;
103  } else if (x == 1) {
104  return PI*PI/6;
105  } else if (x < 2) {
106  const double l = std::log(x);
107  y = 1 - 1/x;
108  r = PI*PI/6 - l*(std::log(y) + 0.5*l);
109  s = 1;
110  } else {
111  const double l = std::log(x);
112  y = 1/x;
113  r = PI*PI/3 - 0.5*l*l;
114  s = -1;
115  }
116 
117  const double y2 = y*y;
118  const double y4 = y2*y2;
119  const double p = P[0] + y * P[1] + y2 * (P[2] + y * P[3]) +
120  y4 * (P[4] + y * P[5]);
121  const double q = Q[0] + y * Q[1] + y2 * (Q[2] + y * Q[3]) +
122  y4 * (Q[4] + y * Q[5] + y2 * Q[6]);
123 
124  return r + s*y*p/q;
125 }

127 /**
128  * @brief Complex dilogarithm \f$\mathrm{Li}_2(z)\f$
129  * @param z_ complex argument
130  * @return \f$\mathrm{Li}_2(z)\f$
131  * @note Implementation translated from SPheno to C++
132  * @author Werner Porod
133  * @note translated to C++ by Alexander Voigt
134  */
135 std::complex<double> dilog(const std::complex<double>& z_) noexcept
136 {
137  const double PI = 3.1415926535897932;
138  const Complex<double> z = { std::real(z_), std::imag(z_) };
139 
140  // bf[1..N-1] are the even Bernoulli numbers / (2 n + 1)!
141  // generated by: Table[BernoulliB[2 n]/(2 n + 1)!, {n, 1, 9}]
142  const double bf[10] = {
143  - 1.0/4.0,
144  + 1.0/36.0,
145  - 1.0/3600.0,
146  + 1.0/211680.0,
147  - 1.0/10886400.0,
148  + 1.0/526901760.0,
149  - 4.0647616451442255e-11,
150  + 8.9216910204564526e-13,
151  - 1.9939295860721076e-14,
152  + 4.5189800296199182e-16
153  };
154 
155  // special cases
156  if (z.im == 0) {
157  if (z.re <= 1) {
158  return dilog(z.re);
159  }
160  // z.re > 1
161  return { dilog(z.re), -PI*std::log(z.re) };
162  }
163 
164  const double nz = norm_sqr(z);
165 
166  if (nz < std::numeric_limits<double>::epsilon()) {
167  return z*(1.0 + 0.25*z);
168  }
169 
170  Complex<double> u(0.0, 0.0), rest(0.0, 0.0);
171  double sgn = 1;
172 
173  // transformation to |z|<1, Re(z)<=0.5
174  if (z.re <= 0.5) {
175  if (nz > 1) {
176  const Complex<double> lz = log(-z);
177  u = -log(1.0 - 1.0 / z);
178  rest = -0.5*lz*lz - PI*PI/6;
179  sgn = -1;
180  } else { // nz <= 1
181  u = -log(1.0 - z);
182  rest = 0;
183  sgn = 1;
184  }
185  } else { // z.re > 0.5
186  if (nz <= 2*z.re) {
187  u = -log(z);
188  rest = u*log(1.0 - z) + PI*PI/6;
189  sgn = -1;
190  } else { // nz > 2*z.re
191  const Complex<double> lz = log(-z);
192  u = -log(1.0 - 1.0 / z);
193  rest = -0.5*lz*lz - PI*PI/6;
194  sgn = -1;
195  }
196  }
197 
198  const Complex<double> u2(u*u);
199 
200  return sgn*(u + u2*(bf[0] + u*horner<1>(u2, bf))) + rest;
201 }

203 /**
204  * @brief Clausen function \f$\mathrm{Cl}_2(\theta) = \mathrm{Im}(\mathrm{Li}_2(e^{i\theta}))\f$
205  * @param x real angle
206  * @return \f$\mathrm{Cl}_2(\theta)\f$
207  * @author Alexander Voigt
208  * @note Implemented as economized Padé approximation.
209  */
210 double clausen_2(double x) noexcept
211 {
212  const double PI = 3.14159265358979324;
213  const double PI2 = 2*PI, PIH = PI/2, PI28 = PI*PI/8;
214  double sgn = 1;
215 
216  if (x < 0) {
217  x = -x;
218  sgn = -1;
219  }
220 
221  if (x >= PI2) {
222  x = std::fmod(x, PI2);
223  }
224 
225  if (x > PI) {
226  const double p0 = 6.28125;
227  const double p1 = 0.0019353071795864769253;
228  x = (p0 - x) + p1;
229  sgn = -sgn;
230  }
231 
232  if (x == 0 || x == PI) {
233  return 0;
234  }
235 
236  double h = 0;
237 
238  if (x < PIH) {
239  const double P[] = {
240  1.3888888888888889e-02, -4.3286930203743071e-04,
241  3.2779814789973427e-06, -3.6001540369575084e-09
242  };
243  const double Q[] = {
244  1.0000000000000000e+00, -3.6166589746694121e-02,
245  3.6015827281202639e-04, -8.3646182842184428e-07
246  };
247  const double y = x*x;
248  const double y2 = y*y;
249  const double p = P[0] + y * P[1] + y2 * (P[2] + y * P[3]);
250  const double q = Q[0] + y * Q[1] + y2 * (Q[2] + y * Q[3]);
251 
252  h = x*(1 - std::log(x) + y*p/q);
253  } else {
254  const double P[] = {
255  6.4005702446195512e-01, -2.0641655351338783e-01,
256  2.4175305223497718e-02, -1.2355955287855728e-03,
257  2.5649833551291124e-05, -1.4783829128773320e-07
258  };
259  const double Q[] = {
260  1.0000000000000000e+00, -2.5299102015666356e-01,
261  2.2148751048467057e-02, -7.8183920462457496e-04,
262  9.5432542196310670e-06, -1.8184302880448247e-08
263  };
264  const double y = PI - x;
265  const double z = y*y - PI28;
266  const double z2 = z*z;
267  const double z4 = z2*z2;
268  const double p = P[0] + z * P[1] + z2 * (P[2] + z * P[3]) +
269  z4 * (P[4] + z * P[5]);
270  const double q = Q[0] + z * Q[1] + z2 * (Q[2] + z * Q[3]) +
271  z4 * (Q[4] + z * Q[5]);
272 
273  h = y*p/q;
274  }
275 
276  return sgn*h;
277 }
